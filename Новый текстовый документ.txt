import "dart:math"; 
void main(List<String> arguments) 
{ 
int min = -10; 
int max = 10; 
int minValue = 0; 
List<int> nums = List.generate(10, (w) => min + Random().nextInt(max - min)); 
print(nums.join(", ")); 
for (int i = 0; i < 10; i++) 
{ 
if (nums[i] < minValue) 
{ 
minValue = nums[i]; 
} 
} 
print("$minValue"); 
}
